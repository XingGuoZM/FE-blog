class Observer{
  static events = new Map();
  static on = (name,fn)=>{
    this.events.set(name,{isOnce:false,fn});
  }
  static once=(name,fn)=>{
    this.events.set(name,{isOnce:true,fn});
  }
  static off = (name) =>{
    this.events.delete(name)
  }
  static emit=(name,data)=>{
    let cashe = this.events.get(name);
    if(cashe){
      if(cashe.isOnce){
        this.events.delete(name)
      }
      cashe.fn(data);
    }
  }
}

  return new Promise((resolve,reject)=>{
    const ans =[];
    let count=0;

          reject(err);
